CREATE TABLE Student (
	id INT NOT NULL IDENTITY(1,1),
	full_name VARCHAR(200) NOT NULL,
        date_of_birth DATE NOT NULL,
	address varchar(200) DEFAULT NULL,
	email varchar(100) DEFAULT NULL,
	group_id INT NOT NULL,
	budget INT NOT NULL,
	PRIMARY KEY (id)
);


CREATE TABLE Groups (
    	id INT NOT NULL IDENTITY(1,1),
    	group_names VARCHAR(10) NOT NULL,
    	derect_id INT NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE dir_stud (
    	id INT NOT NULL IDENTITY(1,1),
    	dir_name VARCHAR(200) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE courses (
    	id INT NOT NULL IDENTITY(1,1),
    	name VARCHAR(200) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE prepodovatyely (
    	id INT NOT NULL IDENTITY(1,1),
    	name VARCHAR(200) NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE dircoprepod (
    	id INT NOT NULL IDENTITY(1,1),
    	derect_id INT NOT NULL,
    	courses_id INT NOT NULL,
    	teacher_id INT NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Marks (
    	id INT NOT NULL IDENTITY(1,1),
    	student_id INT NOT NULL,
    	subject_co_id INT NOT NULL,
    	mark INT,
    	PRIMARY KEY (id)
);

CREATE TABLE Pairs_Time (
    	id INT NOT NULL IDENTITY(1,1),
    	time_start TIME NOT NULL,
    	time_end TIME NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Les_Time (
    	id INT NOT NULL IDENTITY(1,1),
    	subject_co_id INT NOT NULL,
    	time_id INT NOT NULL,
    	date DATE NOT NULL,
    	PRIMARY KEY (id)
);

CREATE TABLE Posechaem (
    	id INT NOT NULL IDENTITY(1,1),
    	timing_id INT NOT NULL,
    	student_id INT NOT NULL,
    	presense BIT NOT NULL,
    	PRIMARY KEY (id)
);

ALTER TABLE Student ADD CONSTRAINT Student_fk0 FOREIGN KEY (group_id) REFERENCES Groups(id);

ALTER TABLE telephone ADD CONSTRAINT telephone_fk0 FOREIGN KEY (student_id) REFERENCES Student(id);

ALTER TABLE Groups ADD CONSTRAINT Groups_fk0 FOREIGN KEY (derect_id) REFERENCES dir_stud(id);

ALTER TABLE dircoprepod ADD CONSTRAINT dircoprepod_fk0 FOREIGN KEY (derect_id) REFERENCES dir_stud(id);

ALTER TABLE dircoprepod ADD CONSTRAINT dircoprepod_fk1 FOREIGN KEY (courses_id) REFERENCES courses(id);

ALTER TABLE dircoprepod ADD CONSTRAINT dircoprepod_fk2 FOREIGN KEY (teacher_id) REFERENCES prepodovatyely(id);

ALTER TABLE Marks ADD CONSTRAINT Marks_fk0 FOREIGN KEY (student_id) REFERENCES Student(id);

ALTER TABLE Marks ADD CONSTRAINT Marks_fk1 FOREIGN KEY (subject_co_id) REFERENCES dircoprepod(id);

ALTER TABLE Les_Time ADD CONSTRAINT Les_Time_fk0 FOREIGN KEY (subject_co_id) REFERENCES dircoprepod(id);

ALTER TABLE Les_Time ADD CONSTRAINT Les_Time_fk1 FOREIGN KEY (time_id) REFERENCES Pairs_Time(id);

ALTER TABLE Posechaem ADD CONSTRAINT Posechaem_fk0 FOREIGN KEY (timing_id) REFERENCES Les_Time(id);
